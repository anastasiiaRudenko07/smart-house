class SmartHouse {
    properties:
        #name: String
        #devicesList: Map<Number, Device>
    behavior:
        +constructor(name: String): void
        
        +get name(): String
        +set name(name: String): void

        -validateDeviceByName(name: String): Bollean      
        -generateId(): Number
        
        +addDevice(device: Device): void
        +get devicesList(): String[]
        +getIdOfDevice(device: Device): Number
        +getDeviceById(id: Number): device
        +deleteDeviceById(id: Number): void
        +deleteAllDevices(): void
        
        +onAllDevices(): void
        +offAllDevices(): void
 }
 
 class Device {
    properties:
        #state: Boolean
        #name: String
        #temperature: Number
    behavior:
        +constructor(name: String): void
        +on(): void
        +off(): void
        +get state(): Boolean
        +get name(): String
        +get temperature: Number
 }

        +handleSteamer(): void
        +get steamerOn(): Boolean
 
        +get waterLevel(): Number

 }
